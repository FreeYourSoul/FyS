35 [[nodiscard]] static SideOrient
36 convertArenaOrientToFb(HexagonSide::Orientation orient)
37 {
38  switch (orient) {
39  case HexagonSide::Orientation::A_N: return SideOrient_A_N;
40  case HexagonSide::Orientation::A_NE: return SideOrient_A_NE;
41  case HexagonSide::Orientation::A_NW: return SideOrient_A_NW;
42  case HexagonSide::Orientation::A_S: return SideOrient_A_S;
43  case HexagonSide::Orientation::A_SE: return SideOrient_A_SE;
44  case HexagonSide::Orientation::A_SW: return SideOrient_A_SW;
45 
46  case HexagonSide::Orientation::B_N: return SideOrient_B_N;
47  case HexagonSide::Orientation::B_NE: return SideOrient_B_NE;
48  case HexagonSide::Orientation::B_NW: return SideOrient_B_NW;
49  case HexagonSide::Orientation::B_S: return SideOrient_B_S;
50  case HexagonSide::Orientation::B_SE: return SideOrient_B_SE;
51  case HexagonSide::Orientation::B_SW: return SideOrient_B_SW;
52 
53  case HexagonSide::Orientation::C_N: return SideOrient_C_N;
54  case HexagonSide::Orientation::C_NE: return SideOrient_C_NE;
55  case HexagonSide::Orientation::C_NW: return SideOrient_C_NW;
56  case HexagonSide::Orientation::C_S: return SideOrient_C_S;
57  case HexagonSide::Orientation::C_SE: return SideOrient_C_SE;
58  case HexagonSide::Orientation::C_SW: return SideOrient_C_SW;
59 
60  case HexagonSide::Orientation::NONE: break;
61  }
62  SPDLOG_ERROR("Trying to convert Orientation::NONE, value replaced by {}", SideOrient_MIN);
63  return SideOrient_MIN;
64 }
