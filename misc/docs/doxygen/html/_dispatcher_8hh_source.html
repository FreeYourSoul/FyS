84  void runDispatching() {

86  auto [listenerSocketHasSomethingToPoll,
87  subscriberSocketHasSomethingToPoll,
88  dispatcherSocketHasSomethingToPoll] = _connectionManager.poll();
89  if (listenerSocketHasSomethingToPoll) {
90  _connectionManager.dispatchMessageFromListenerSocket(
91  [this](zmq::multipart_t &&msg, network::DispatcherConnectionManager &manager) {
92  _dispatcher.processInputMessage(std::move(msg), manager);

95  }
96  if (subscriberSocketHasSomethingToPoll) {
97  _connectionManager.dispatchMessageFromSubscriberSocket(
98  [this](zmq::multipart_t &&msg, network::DispatcherConnectionManager &manager) {
99  _dispatcher.processClusterMessage(std::move(msg), manager);
100  }
101  );
102  }
103  if (dispatcherSocketHasSomethingToPoll) {
104  _connectionManager.dispatchMessageFromDispatcherSocket(
105  [this](zmq::multipart_t && msg, network::DispatcherConnectionManager &manager){
106  _dispatcher.forwardMessageToFrontend(std::move(msg), manager);

108  );
109  }
110  }
