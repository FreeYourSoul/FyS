37 [[nodiscard]] inline std::string
38 getActionNameFromKey(const std::string& key)
39 {
40  auto startSeparator = key.find_last_of(':');
41  auto endSeparator = key.find_last_of('.');
42 
43  if (startSeparator != std::string::npos && endSeparator != std::string::npos) {
44  // if a ':' and a '.' are found
45  return key.substr(startSeparator + 1, endSeparator - startSeparator - 1);
46  }
47  else if (startSeparator == std::string::npos && endSeparator != std::string::npos) {
48  // if a ':' is not found but not a '.' is found
49  return key.substr(0, endSeparator);
50  }
51  else if (startSeparator != std::string::npos && endSeparator == std::string::npos) {
52  // if a ':' is found but not a '.' is not found
53  return key.substr(startSeparator + 1);
54  }
55  return key;
56 }

195 static void
196 mergeAlterations(std::vector<Alteration>& toModify, std::vector<Alteration> toMerge, bool replaceIfExist = false)
197 {
198  (void)mergeAlterations; // suppress unused warning (as it is used, but by ChaiScript)
199 
200  std::move(toMerge.begin(), toMerge.end(), std::back_inserter(toModify));
201  std::sort(toModify.begin(), toModify.end(), [replaceIfExist](const Alteration& lhs, const Alteration& rhs) {
202  return lhs.getAlterationKey() == rhs.getAlterationKey() &&
203  ((replaceIfExist && lhs.getTurn() > rhs.getTurn()) || (!replaceIfExist && lhs.getTurn() < rhs.getTurn()));
204  });
205  toModify.erase(std::unique(toModify.begin(), toModify.end(), [](const Alteration& lhs, const Alteration& rhs) {
206  return lhs.getAlterationKey() == rhs.getAlterationKey();
207  }), toModify.end());
208 }
