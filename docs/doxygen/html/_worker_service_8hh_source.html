115  template<typename HandlerAuth, typename HandlerInGame>
116  void pollAndProcessMessageFromPlayer(HandlerAuth&& handlerAuth, HandlerInGame&& handlerInGame) noexcept
117  {
118  // Initialize poll set
119  zmq::pollitem_t items[] = {
120  {_workerRouter, 0, ZMQ_POLLIN, 0}
121  };
122  zmq::poll(&items[0], 1, 100);
123  if (static_cast<bool>(items[0].revents & ZMQ_POLLIN)) {
124  zmq::multipart_t msg;
125  if (!msg.recv(_workerRouter, ZMQ_NOBLOCK)) {
126  SPDLOG_ERROR("Error while reading on the arena worker listener socket");
127  }
128  else if (auto s = msg.size(); s != 3) {
129  SPDLOG_WARN("An incoming message with {} instead of 3 has been read", s);
130  }
131  else {
132  auto identity = msg.pop();
133  auto intermediate = msg.pop();
134  auto content = msg.pop();
135  if ("auth" == intermediate.to_string_view()) {
136  std::forward<HandlerAuth>(handlerAuth)(std::move(identity), std::move(content));
137  }
138  else {
139  std::forward<HandlerInGame>(handlerInGame)(std::move(identity), intermediate, std::move(content));
140  }
141  }
142  }
143  }
